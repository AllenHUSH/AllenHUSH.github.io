# 清除浮动

# 为什么要清除浮动

  • 背景不能显示
  • 边框不能撑开
  • margin不能正确显示

# 清除浮动的方法

# 添加新的元素 、应用 clear:both;

<div class="outer">
    <div class="float">1</div>
    <div class="float">2</div>
    <div class="float">3</div>
    <div class="clear"></div>
</div>

.clear{clear:both; height: 0; line-height: 0; font-size: 0}

# 父级div定义overflow:auto

<div class="outer"> //这里添加了一个class
    <div class="float">1</div>
    <div class="float">2</div>
    <div class="float">3</div>
</div>

.outer{
    overflow: auto; 
  	zoom: 1; //ie专属 处理兼容性问题
}

# 父元素伪元素after

.outer {
  zoom:1;
}    /*==for IE6/7 Maxthon2==*/
.outer ::after {
	clear:both;
	content:'.';
	display:block;
	width: 0;
	height: 0;
	visibility:
	hidden;
}/*==for FF/chrome/opera/IE8==*/
